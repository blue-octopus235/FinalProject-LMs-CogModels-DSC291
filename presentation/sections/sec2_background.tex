  \begin{itemize}
  %  \item TODO

\item \textbf{Hierarchical rule}: agree with the true subject.
\item \textbf{Linear rule}: agree with the noun closest to the verb.
\end{itemize}


\textbf{Example}

\begin{quote}
The keys to the cabinet \underline{\hspace{1cm}} on the table.
\end{quote}

\begin{itemize}
\item True subject = \textit{keys} 
\item Nearest noun = \textit{cabinet} 
\end{itemize}

\begin{itemize}
\item Hierarchical rule $\Rightarrow$ \textbf{are}
\item Linear heuristic $\Rightarrow$ \textbf{is}
\end{itemize}


\textbf{Key idea:} SVA creates cases where the two rules make different predictions.

\end{frame}

\begin{frame}{Architectural Inductive Biases}

\textbf{Two competing learners}

\vspace{0.2cm}

\begin{columns}[T]

\column{0.48\textwidth}

\textbf{LSTM}

\begin{itemize}
    \item Sequential recurrent model
    \item Processes sentences word by word
    \item No explicit syntactic structure
    \item Learns grammar from statistical regularities
\end{itemize}

\vspace{0.2cm}

\centering
\textbf{Weak structural prior}

\column{0.48\textwidth}

\textbf{RNNG}

\begin{itemize}
    \item Generates sentences and parse trees jointly
    \item Explicit hierarchical representations
    \item Encodes phrase structure directly
    \item Strong syntactic inductive bias
\end{itemize}

\vspace{0.2cm}

\centering
\textbf{Strong structural prior}

\end{columns}

\vspace{0.5cm}

\textbf{Key Question}

\begin{center}
Does explicit syntactic structure help preserve hierarchical grammatical rules under noisy input?
\end{center}

\end{frame}

\begin{frame}[allowframebreaks]{Related Work and Research Gap}

\textbf{SVA as a syntax benchmark}

\begin{itemize}
\item Linzen (2016): LSTMs learn agreement but struggle with attractors.
\item Gulordava (2018): evidence for abstract syntactic generalization.
\end{itemize}

\vspace{0.2cm}

\textbf{Structural inductive bias}

\begin{itemize}
\item Dyer (2016): Recurrent Neural Network Grammar (RNNG).
\item Kuncoro (2018): explicit syntax improves long-distance dependencies.
\end{itemize}

\vspace{0.2cm}

\begin{itemize}
\item Prior work studies noisy inputs and grammatical corruption.
\item Focus is typically on downstream performance.
\end{itemize}

\textbf{Research gap}

\begin{itemize}
\item Little work asks whether grammatical rules themselves can still be acquired under noisy supervision.
\item No systematic study of how SVA corruption affects:
\begin{itemize}
\item rule acquisition,
\item hierarchical vs. linear generalization,
\item LSTM vs. RNNG robustness.
\end{itemize}
\end{itemize}

\end{frame}
