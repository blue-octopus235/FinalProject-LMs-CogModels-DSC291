\section{Background}

\begin{frame}{Subject--verb agreement: hierarchical vs.\ linear}
  \begin{itemize}
    \item \textbf{Hierarchical rule}: agree with the true syntactic subject.
    \item \textbf{Linear rule}: agree with the noun closest to the verb.
  \end{itemize}

  \vspace{0.3cm}
  \textbf{Example}
  \begin{quote}
    The keys to the cabinet \underline{\hspace{1cm}} on the table.
  \end{quote}

  \begin{columns}[T]
    \column{0.5\textwidth}
    \begin{itemize}
      \item True subject $=$ \textit{keys} (plural)
      \item Nearest noun $=$ \textit{cabinet} (singular)
    \end{itemize}
    \column{0.5\textwidth}
    \begin{itemize}
      \item Hierarchical $\Rightarrow$ \textbf{are}
      \item Linear $\Rightarrow$ \textbf{is}
    \end{itemize}
  \end{columns}

  \vspace{0.3cm}
  \textbf{Key idea:} SVA with attractors creates cases where the two rules
  make \emph{different} predictions. This is our diagnostic lever.
\end{frame}

\begin{frame}{Attractors as the diagnostic case}
  \begin{itemize}
    \item An \textbf{attractor} is an intervening noun that mismatches the
          subject in number (e.g.\ \textit{cabinet} between
          \textit{keys} and the verb).
    \item On \textbf{no-attractor} items, both rules agree
          $\Rightarrow$ accuracy is high for either strategy.
    \item On \textbf{attractor} items, only the hierarchical rule is correct
          $\Rightarrow$ success requires ignoring the misleading linear cue.
    \item The \textbf{attractor gap} (no-attractor acc $-$ attractor acc)
          measures how much the model relies on the linear shortcut.
  \end{itemize}
\end{frame}

\begin{frame}{Architectural inductive biases}
  \textbf{Two competing learners}

  \vspace{0.2cm}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{LSTM}
    \begin{itemize}
      \item Sequential recurrent model
      \item No explicit syntactic structure
      \item Learns grammar from statistical regularities
    \end{itemize}
    \centering\vspace{0.2cm}
    \textbf{Weak structural prior}

    \column{0.48\textwidth}
    \textbf{RNNG}
    \begin{itemize}
      \item Generates sentences and parse trees jointly
      \item Explicit hierarchical representations
      \item Strong syntactic inductive bias
    \end{itemize}
    \centering\vspace{0.2cm}
    \textbf{Strong structural prior}
  \end{columns}

  \vspace{0.4cm}
  \begin{center}
    \textbf{Key question:} Does explicit syntactic structure help preserve
    hierarchical rules under noisy input?
  \end{center}
\end{frame}

\begin{frame}{Related work and research gap}
  \textbf{SVA as a syntax benchmark}
  \begin{itemize}
    \item Linzen et al.\ (2016): LSTMs learn agreement but struggle with attractors.
    \item Gulordava et al.\ (2018): evidence for abstract syntactic generalization.
  \end{itemize}

  \vspace{0.2cm}
  \textbf{Structural inductive bias}
  \begin{itemize}
    \item Dyer et al.\ (2016): Recurrent Neural Network Grammar (RNNG).
    \item Kuncoro et al.\ (2018): explicit syntax improves long-distance dependencies.
  \end{itemize}

  \vspace{0.2cm}
  \textbf{Research gap}
  \begin{itemize}
    \item Prior work tests models on clean data; little asks whether
          grammatical rules can still be \emph{acquired} under noisy supervision.
    \item No systematic study of how SVA corruption affects rule acquisition,
          hierarchical vs.\ linear generalization, or LSTM vs.\ RNNG robustness.
  \end{itemize}
\end{frame}
